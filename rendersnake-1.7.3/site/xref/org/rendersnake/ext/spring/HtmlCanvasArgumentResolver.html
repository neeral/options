52      static void setupPageContext(HttpServletRequest request, PageContext context) {
53          String q = request.getQueryString();
54          context.withObject(PageContext.REQUEST_PATH, request.getPathTranslated());
55          context.withObject(PageContext.REQUEST_URIQ, request.getRequestURI() + (q == null ? "" : "?" + q));
56          context.withObject(PageContext.SESSION, new SessionAttributesMap(request));
57  		context.withObject(PageContext.REQUEST_PARAMETERS, new RequestParametersMap(request));	
58          context.withObject(PageContext.REQUEST_HEADERS, new RequestHeadersMap(request));
